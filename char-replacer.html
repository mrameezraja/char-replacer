Character Replacer :

        //Default Behaviour
        $(function () {
            $("#Text1").charReplacer();
        })
        
        $(function () {
            $("#Text2").charReplacer({ replaceWith: '$' });
        })
        
        $(function () {
           $("#Text3").charReplacer({ replaceChar: '&', replaceWith: 'and' });
        })
        
        $(function () {
           $("#Text4").charReplacer({ replaceChar: '\'', replaceWith: '\"' });
        })
        
        $(function () {
            $("#Text5").charReplacer({ replaceChar: ["<", ">", "\n"], replaceWith: ["&lt;", "&gt;", "<br>"] });
        })
        
        // mutiple fields
        //html
        <input type='text' char-replace='true' />
        <input type='text' char-replace='true' />    
        <input type='text' char-replace='true' />
        <input type='text' char-replace='true' />
        // javascript
        $(function () {
           $("input[char-replace='true']").charReplacer();
           //or
           //$("input[char-replace='true']").charReplacer({ replaceChar:" ", replaceWith:"-" });
        })
